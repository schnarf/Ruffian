def factorial ( int n ) -> int {
	if( n == 0 ) { return 1; }
	return n * factorial(n - 1);
}

def fibonacci( int n ) -> int {
	if( n < 3 ) { return 1; }
	return fibonacci(n-1) + fibonacci(n-2);
}

def decl_test(int n) -> int {
	int ret;
	if( n < 3 ) {
		int x= factorial(n);
		ret= x;
	} else {
		int x= fibonacci(n);
		ret= x;
	}

	return ret;
}

def factorial ( int n ) -> int {
	if( n == 0 ) { return 1; }
	return n * factorial(n - 1);
}

def fibonacci( int n ) -> int {
	if( n < 3 ) { return 1; }
	return fibonacci(n-1) + fibonacci(n-2);
}

def decl_test(int n) -> int {
	int ret;
	if( n < 3 ) {
		int x= factorial(n);
		ret= x;
	} else {
		int x= fibonacci(n);
		ret= x;
	}

	return ret;
}

def insertion_sort( int[] a, int n ) {
	for( int i=0; i<n; ++i ) {
		int min= i;
		for( int j=i+1; j<n; ++j ) {
			if( a[j] < a[min] ) { min= j; }
		}
		int temp= a[i];
		a[i]= a[min];
		a[min]= temp;
	}
}

// Prototypes for standard library functions
def printInt(int n);
def printFloat(float f);
def printDouble(double f);
def printNewline();
def rand() -> int;

def main(int n) -> int {
	for( int i=0; i<4; ++i ) {
		printInt(i);
	}

	int b= 4;
	foreach( int i; 0 : b ) {
		printInt(i);
		b= b * 2;
	}
	printInt(b);

	return 0;
}
